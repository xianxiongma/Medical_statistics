# 转换数据
row2df <- function(x){
  group <- rep(as.numeric(x[1]), x[3])
  conc <- rep(as.numeric(x[2]), x[3])
  return(data.frame(group=group, conc=conc))
}
​
test_data <- plyr::adply(ordinal_data, .margins=1, row2df)[, c(-1, -2, -3)]
​
# 秩和检验
wilcox.test(conc~group, data=test_data, exact=F)
